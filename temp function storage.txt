def check_collision(p1,p2):
        target = pr.Vector3(
                                    p2.position.x - p1.position.x,
                                    p2.position.y - p1.position.y,
                                    p2.position.z - p1.position.z 
                                    )
            
        r = math.sqrt(target.x**2 + target.y**2 + target.z**2)
        if r <= p1.radius + p2.radius:
            return True
        else:
            return False
